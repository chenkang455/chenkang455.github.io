def f(x):
    return 1-x*pow(2.718, -x*x)
# parameters define
left = 0
right = 1
gold_parameter = 0.618
# make graph

y_numbers = [f(x) for x in x_numbers]
plt.plot(x_numbers, y_numbers)
iter = 0
# iteration 
while right - left > 0.01:
    length = right - left
    x1 = left + (1-gold_parameter)*length
    x2 = left + gold_parameter*length
    x = [x1, x2]
    y = [f(x1), f(x2)]
    plt.scatter(x, y)
    # judge
    if f(x1) > f(x2):
        left = x1
    else:
        right = x2
    iter = iter + 1
